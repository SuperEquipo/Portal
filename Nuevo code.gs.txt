function doGet(e) {
  if (e && e.parameter && e.parameter.id) {
    var id = e.parameter.id;
    
    try {
      // 1. PRIMERA COMPROBACIÓN: ¿Es el archivo de la Presentación?
      var archivo = DriveApp.getFileById(id);
      var nombreArchivo = archivo.getName().toLowerCase();
      
      if (nombreArchivo.includes("presentación") || nombreArchivo.includes("presentacion") || archivo.getMimeType() === MimeType.HTML) {
        var contenidoHtml = archivo.getBlob().getDataAsString();
        return HtmlService.createHtmlOutput(contenidoHtml)
          .setTitle("Tu Plan Maestro - Súper Equipo")
          .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL);
      }
    } catch(errArchivo) {
      // Si lanza error aquí, significa que el ID no es de un archivo, sino de la CARPETA de la App.
    }

    try {
      // 2. SEGUNDA COMPROBACIÓN: Cargar la App principal (Carpeta)
      var carpetaCliente = DriveApp.getFolderById(id);
      var nombreCliente = carpetaCliente.getName().split("(")[0].trim();
      var datosHTML = obtenerDatosParaLaApp(id, nombreCliente);
      
      var plantilla = HtmlService.createTemplateFromFile('AppCliente');
      plantilla.datos = datosHTML;
      
      return plantilla.evaluate()
        .setTitle("Súper Equipo - " + nombreCliente)
        .addMetaTag('viewport', 'width=device-width, initial-scale=1.0, maximum-scale=1.0, user-scalable=no')
        .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL);
        
    } catch (errCarpeta) {
      return HtmlService.createHtmlOutput(
        "<h2>Error de acceso</h2>" +
        "<p>No se pudo sincronizar su perfil.</p>" +
        "<p><strong>Detalle:</strong> " + errCarpeta.toString() + "</p>"
      );
    }
  } else {
    procesarNuevosClientes();
    return ContentService.createTextOutput("🚀 Sistema de Onboarding procesado con éxito.");
  }
}

function obtenerDatosParaLaApp(carpetaId, nombre) {
  var carpeta = DriveApp.getFolderById(carpetaId);
  var datos = {
    nombre: nombre,
    carpetaId: carpetaId,
    informe: "Aún no se ha generado tu evaluación inicial. Por favor, asegúrate de enviar el cuestionario de entrada.",
    ejercicios: [],
    alimentacion: [],
    formUrl: "",
    presentacionUrl: "",
    videoUrl: "", 
    respuestasEnviadas: false, 
    logoBase64: obtenerLogoBase64() 
  };
  
  var filesDoc = carpeta.getFilesByName("3. Informe de seguimiento");
  if (filesDoc.hasNext()) {
    var doc = DocumentApp.openById(filesDoc.next().getId());
    datos.informe = doc.getBody().getText();
  }
  
  var filesSheet = carpeta.getFilesByName("2. Control y Planificación - " + nombre);
  if (filesSheet.hasNext()) {
    var ss = SpreadsheetApp.openById(filesSheet.next().getId());
    
    var hojaEjercicios = ss.getSheetByName("Ejercicios");
    if (hojaEjercicios) {
      datos.ejercicios = hojaEjercicios.getDataRange().getValues();
    }
    var hojaAlim = ss.getSheetByName("Alimentación");
    if (hojaAlim) {
      datos.alimentacion = hojaAlim.getDataRange().getValues();
    }

    var hojaDatos = ss.getSheetByName("Datos físicos");
    if (hojaDatos && hojaDatos.getLastRow() > 2) {
      datos.respuestasEnviadas = true;
    }
  }

  var archivosEnCarpeta = carpeta.getFiles();
  while (archivosEnCarpeta.hasNext()) {
    var archivo = archivosEnCarpeta.next();
    var nomArchivo = archivo.getName().toLowerCase();
    
    if (archivo.getMimeType() === MimeType.GOOGLE_FORMS) {
      var formFile = FormApp.openById(archivo.getId());
      datos.formUrl = formFile.getPublishedUrl();
      if (formFile.getResponses().length > 0) {
        datos.respuestasEnviadas = true;
      }
    }
    
    if (nomArchivo === "presentación inicial" || nomArchivo === "presentacion inicial" || nomArchivo.includes("estamos preparando") || nomArchivo.startsWith("4.")) {
      datos.presentacionUrl = URL_DE_LA_APP + "?id=" + archivo.getId();
    }

    // Video explicativo (NotebookLM)
    var esVideoPorMime = archivo.getMimeType().startsWith("video/");
    var esVideoPorNombre = nomArchivo.includes("video") || nomArchivo.includes(".mp4") || nomArchivo.includes("notebook");
    if (esVideoPorMime || esVideoPorNombre) {
      archivo.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
      // 🌟 CORRECCIÓN: Mandamos únicamente el ID puro a la App para reconstruir un visor limpio
      datos.videoUrl = archivo.getId();
    }
  }

  return datos;
}

function obtenerLogoBase64() {
  try {
    var carpetaBase = DriveApp.getFolderById(ID_CARPETA_IMAGENES_PROPIAS);
    var archivos = carpetaBase.getFiles();
    while (archivos.hasNext()) {
      var archivo = archivos.next();
      var nombreLimpio = archivo.getName().toLowerCase();
      if (nombreLimpio.startsWith("avatar-inicio")) {
        var bytes = archivo.getBlob().getBytes();
        return Utilities.base64Encode(bytes);
      }
    }
  } catch (e) {
    Logger.log("Error al recuperar el logotipo en base64: " + e.toString());
  }
  return ""; 
}

function subirImagenDesdeApp(base64Data, nombreArchivo, carpetaClienteId) {
  try {
    var carpetaCliente = DriveApp.getFolderById(carpetaClienteId);
    var subFolder = carpetaCliente.getFoldersByName("Seguimiento Báscula");
    var destino = subFolder.hasNext() ? subFolder.next() : carpetaCliente;
    
    var contentType = base64Data.substring(5, base64Data.indexOf(";"));
    var bytes = Utilities.base64Decode(base64Data.split(",")[1]);
    var blob = Utilities.newBlob(bytes, contentType, nombreArchivo);
    
    var archivoCreado = destino.createFile(blob);
    return { exito: true, nombre: archivoCreado.getName() };
  } catch (e) {
    return { exito: false, error: e.toString() };
  }
}

function configurarHojas(ss) {
  var hojas = ss.getSheets();
  var hojaDatos;
  for (var i = 0; i < hojas.length; i++) {
    if (hojas[i].getName().includes("Respuestas") || hojas[i].getName().includes("Form Responses")) {
      hojas[i].setName("Datos físicos");
      hojaDatos = hojas[i];
    }
  }
  if (!hojaDatos) hojaDatos = ss.getSheets()[0];

  ss.insertSheet("Ejercicios");
  ss.insertSheet("Alimentación");
  if (ss.getSheetByName("Hoja 1")) ss.deleteSheet(ss.getSheetByName("Hoja 1"));
  if (ss.getSheetByName("Sheet1")) ss.deleteSheet(ss.getSheetByName("Sheet1"));

  if (hojaDatos.getMaxColumns() < 35) {
    hojaDatos.insertColumnsAfter(hojaDatos.getMaxColumns(), 15);
  }

  hojaDatos.getRange("A1:AC100").setFontFamily("Arial").setFontSize(10);
  hojaDatos.getRange("A1:I1").setBackground("#2C3E50").setFontColor("#FFFFFF").setFontWeight("bold");
  hojaDatos.getRange("B1").setValue("Género");
  hojaDatos.getRange("K1:AC1").merge().setValue("MÉTRICAS DE BÁSCULA INTELIGENTE")
    .setBackground("#27AE60").setFontColor("#FFFFFF").setFontWeight("bold").setHorizontalAlignment("center");

function ejecutarSeguimientoSemanalTotal() {
  var idMaestraLimpio = ID_CARPETA_MAESTRA.trim();
  var carpetaMaestra;
  try {
    carpetaMaestra = DriveApp.getFolderById(idMaestraLimpio);
  } catch (e) {

    return;
  }
  
  var conocimientoCerebro = obtenerConocimientoCerebro();
  var subcarpetas = carpetaMaestra.getFolders();

  while (subcarpetas.hasNext()) {
    var carpetaCliente = subcarpetas.next();
    var nombreCliente = carpetaCliente.getName().split("(")[0].trim();

    try {
      var folderSeguimiento = carpetaCliente.getFoldersByName("Seguimiento Báscula");
      if (!folderSeguimiento.hasNext()) continue;
      var carpetaSeguimiento = folderSeguimiento.next();

      var spreadsheetId = null;
      var documentId = null;

      var filesSpreadsheet = carpetaCliente.getFilesByName("2. Control y Planificación - " + nombreCliente);
      if (filesSpreadsheet.hasNext()) spreadsheetId = filesSpreadsheet.next().getId();

function procesarSeguimientoDeCliente(nombre, folderSeguimiento, spreadsheetId, documentId, conocimientoCerebro) {
  var carpetaProcesadas;
  var subCarpetas = folderSeguimiento.getFoldersByName("Imágenes Procesadas");
  if (subCarpetas.hasNext()) {
    carpetaProcesadas = subCarpetas.next();
  } else {
    carpetaProcesadas = folderSeguimiento.createFolder("Imágenes Procesadas");
  }
  
  var archivos = folderSeguimiento.getFiles();
  var imagenesPendientes = [];
  while (archivos.hasNext()) {
    var archivo = archivos.next();
    if (archivo.getMimeType().indexOf("image/") !== -1) {
      imagenesPendientes.push(archivo);
    }
  }
  if (imagenesPendientes.length === 0) return;

  var base64DataList = [];
  var mimeTypes = [];
  var imageFileIds = [];

  for (var i = 0; i < imagenesPendientes.length; i++) {
    var img = imagenesPendientes[i];
    base64DataList.push(Utilities.base64Encode(img.getBlob().getBytes()));
    mimeTypes.push(img.getBlob().getContentType());
    imageFileIds.push(img.getId());
  }

  var resultadoJsonTexto = procesarImagenesMultimodalesConGemini(base64DataList, mimeTypes);
  if (resultadoJsonTexto && resultadoJsonTexto.indexOf("FALLO") === -1) {
    try {
      var datosConsolidados = JSON.parse(resultadoJsonTexto);
      almacenarDatosEnHojaSegura(datosConsolidados, spreadsheetId);
      actualizarInformeDeSeguimiento(datosConsolidados, documentId, nombre, spreadsheetId, conocimientoCerebro);

      for (var j = 0; j < imageFileIds.length; j++) {
        DriveApp.getFileById(imageFileIds[j]).moveTo(carpetaProcesadas);
      }

    } catch(eError) {
      Logger.log("Error JSON: " + eError.toString());
    }
  }
}

        var doc = DocumentApp.openById(archivo.getId());
        textoBase += "[DOCUMENTO: " + archivo.getName() + "]\n" + doc.getBody().getText() + "\n\n";
      } else if (mimeType === MimeType.PLAIN_TEXT) {
        textoBase += "[DOCUMENTO: " + archivo.getName() + "]\n" + archivo.getBlob().getDataAsString() + "\n\n";
      }
    }
    textoBase += "--- FIN DE LA BASE DE DATOS ---";
  } catch (error) {
    textoBase = "No se pudo cargar la base de datos externa.";
  }
  return textoBase;
}

function limpiarMemoriaFantasma() {
  var triggers = ScriptApp.getProjectTriggers();
  for (var i = 0; i < triggers.length; i++) {
    if (triggers[i].getHandlerFunction() === 'alEnviarFormulario') {
      ScriptApp.deleteTrigger(triggers[i]);
    }
  }
}

function obtenerCatalogoDeImagenes() {
  var carpetaBase = DriveApp.getFolderById(ID_CARPETA_IMAGENES_PROPIAS);
  var subcarpetas = carpetaBase.getFolders();
  var catalogo = {};
  var listaNombres = [];

  while (subcarpetas.hasNext()) {
    var sub = subcarpetas.next();
    var archivos = sub.getFiles();
    while (archivos.hasNext()) {
      var archivo = archivos.next();
      var nombreLimpio = archivo.getName().split(".")[0].toLowerCase().trim();
      catalogo[nombreLimpio] = archivo.getId();
      listaNombres.push(nombreLimpio);
    }
  }
  return { mapa: catalogo, texto: listaNombres.join(", ") };
}

function extraerBloque(texto, etiqueta) {
  if (!texto) return "";
  var regExp = new RegExp("<" + etiqueta + ">([\\s\\S]*?)</" + etiqueta + ">", "i");
  var coincidencia = texto.match(regExp);
  return coincidencia ? coincidencia[1].trim() : "";
}
